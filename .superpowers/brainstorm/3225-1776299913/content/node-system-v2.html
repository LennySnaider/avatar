Execution Flow

Client-Side Execution Engine
// 1. User clicks "Run"
sorted = topologicalSort(nodes, edges)

// 2. Execute each node sequentially
for node of sorted:
setNodeStatus(node, 'running') // UI: glow + spinner icon
inputs = getInputsFromPreviousNodes()
result = await handlers[node.type](node, inputs)
context[node.id] = result.output
setNodeStatus(node, result.success ? 'completed' : 'error')

// 3. Handlers map directly to existing services:
'generate-image' → GeminiService.generateAvatar()
'generate-video' → KlingService.generateVideo()
'prompt-enhance' → GeminiService.enhancePrompt()
'describe-image' → GeminiService.describeImageForPrompt()
'script-generator' → ScriptService.generateScript()
'text-to-speech' → MiniMaxService.textToSpeech()
'stitch' → VideoStitchService.stitchVideos()
